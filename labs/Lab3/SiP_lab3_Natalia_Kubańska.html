Wyznacz rozkładd x (w ramce ddanych) oraz stwórz wykres tego rozkładu oraz jego dystrybuante. Można wykorzystać funkcję ecdf(). Wskazówka - korzytając z ecdf należy odpowiedio przygotować dane (próbkę) bazując na prawdopodbieństwie. Funkcja ecdf() to funckja wyznaczająca dystry- buantę empiryczną czy taką, która bazuje na próbce jaką mamy. Jest ona przybliżeniem dystrybuanty teoretycznej.

N = 8
n = 5
X = np.arange(5,
              9)  #losujemy 5 kul bez zwracania więc najmniejszy numer kuli to może być 5 czyli największy numer kuli może być od 5 do 8

total_combinations = comb(N, n)
probabilities = comb(X - 1, n - 1) / total_combinations

df = pd.DataFrame({
    'X': X,
    'P(X=x)': probabilities,
    'CDF': np.cumsum(probabilities)
})

Narysować dystrybuanty oraz rozkłady teoretyczne i empiryczne dla rozkładów dyskretnych oraz dla rozkładów ciągłych. Następnie na podanej próbce oblicz średnią i wariancje dla każdego rozkładu i sprawdź czy zgadzają sie one z wartościami teoretycznymi.

np.random.seed(42)
N = 1000
rozklady = {
    'Dwumianowy (n=10, p=0.5)': {'typ': 'dyskretny', 'dist': binom(n=10, p=0.5)},
    'Poissona (mu=3)': {'typ': 'dyskretny', 'dist': poisson(mu=3)},
    'Geometryczny (p=0.3)': {'typ': 'dyskretny', 'dist': geom(p=0.3)},
    'Normalny (mu=0, sig=1)': {'typ': 'ciagly', 'dist': norm(loc=0, scale=1)},
    'Wykładniczy (scale=2)': {'typ': 'ciagly', 'dist': expon(scale=2)},
    'Jednostajny (0 do 10)': {'typ': 'ciagly', 'dist': uniform(loc=0, scale=10)}
}

wyniki = []
fig, axes = plt.subplots(len(rozklady), 2, figsize=(12, 4 * len(rozklady)))

for i, (nazwa, info) in enumerate(rozklady.items()):
    dist = info['dist']
    typ = info['typ']

    probka = dist.rvs(size=N)

    mean_teo, var_teo = dist.stats(moments='mv')
    mean_emp = np.mean(probka)
    var_emp = np.var(probka, ddof=1)

    wyniki.append({
        'Rozkład': nazwa,
        'Śr. Teoretyczna': float(mean_teo),
        'Śr. Empiryczna': mean_emp,
        'War. Teoretyczna': float(var_teo),
        'War. Empiryczna': var_emp
    })

    ax_pdf = axes[i, 0]
    ax_cdf = axes[i, 1]

    x_min, x_max = probka.min(), probka.max()

    if typ == 'dyskretny':
        x = np.arange(x_min, x_max + 1)

        ax_pdf.plot(x, dist.pmf(x), 'ro', ms=8, label='Teoria (PMF)')
        ax_pdf.vlines(x, 0, dist.pmf(x), colors='r', lw=5, alpha=0.5)
        ax_cdf.step(x, dist.cdf(x), where='post', color='r', label='Teoria (CDF)', lw=2)
        sns.histplot(probka, stat='proportion', discrete=True, color='blue', alpha=0.4, ax=ax_pdf, label='Empiria')
        sns.ecdfplot(probka, ax=ax_cdf, color='blue', linestyle='--', label='Empiria (ECDF)')

    else:
        x = np.linspace(x_min - 1, x_max + 1, 200)

        ax_pdf.plot(x, dist.pdf(x), 'r-', lw=2, label='Teoria (PDF)')
        ax_cdf.plot(x, dist.cdf(x), 'r-', lw=2, label='Teoria (CDF)')

        sns.histplot(probka, stat='density', bins=30, color='blue', alpha=0.4, ax=ax_pdf, label='Empiria')
        sns.ecdfplot(probka, ax=ax_cdf, color='blue', linestyle='--', label='Empiria (ECDF)')

    ax_pdf.set_title(f'{nazwa} - Gęstość/PMF')
    ax_pdf.legend()
    ax_cdf.set_title(f'{nazwa} - Dystrybuanta')
    ax_cdf.legend()

plt.tight_layout()
plt.show()

df_wyniki = pd.DataFrame(wyniki).round(4)
